-- 반복문과 증감 연산자 --

fun main(){
    // 조건형 반복문
    var a = 5
    while(a > 10){
        
    }
    
    do{
        
    }while( a > 10)
    // 범위형 반복문
    for(i in 0..9 step 3) { // i에 var을 붙이지 않아도 된다.
        // 0..9 는 0보다 크거나 같고, 9보다 작거나 같게 반복한다는 것이고 default는 1씩 증가한다.
        // 증가하는 공차를 1이 아닌 다른 것으로 설정하기 위해 step 옵션을 준다. 
        // 감소는 .. 대신에 downTo를 사용한다.
        // i in 9 downTo 0
        // 숫자가 아니라 'a'..'e' 처럼 문자도 반복 가능하다.
        print(i)
    }
}
